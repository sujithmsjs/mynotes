import java.util.Arrays;

public class BubbleSort {

	public static void main(String[] args) {

		//int[] arr = { 23, 1, 4, 2, 6, 7, 3, 10, 4, 1, 4, -23 };
		int[] arr = {-23, 1, 1, 2, 3, 4, 4, 4, 6, 7, 10, 23};
		int[] sort = bubbleSort(arr);
		//System.out.println(Arrays.toString(sort));

	}

	private static int[] bubbleSort(int[] arr) {

		for (int i = 0; i < arr.length; i++) {
			boolean swapped = false;
			for (int j = 0; j < arr.length -1-i; j++) {
				if(arr[j] > arr[j+1]) {
					System.out.println("swapped:)");
					int temp = arr[j];
					arr[j] = arr[j+1];
					arr[j+1] = temp;
					swapped = true;
				}
				System.out.println(Arrays.toString(arr));
			}
			if(! swapped){
				break;
			}
			
			System.out.println();
		}

		return arr;
	}

}

import java.util.Arrays;

public class Sort {

	public static void main(String[] args) {

		int[] arr = {23,1,4,2,6,7,3,10};
		int[] sort = sortIt(arr);
		System.out.println("Sorted values: "+Arrays.asList(arr));
		for (int i = 0; i < sort.length; i++) {
			System.out.println(sort[i]);
		}

		
	}

	private static int[] sortIt(int[] arr) {
		
		System.out.println("Array len: "+arr.length);
		for (int i = 0; i < arr.length; i++) {
			
			//Find minimun values
			int min = Integer.MAX_VALUE;
			int index = i;
			for (int j = i; j < arr.length; j++) {
				
				if(arr[j] < min) {
					min = arr[j];
					index = j;
				}
			}
			//Swap
			int temp = arr[index];
			arr[index] = arr[i];
			arr[i] = temp;
			
			
		}

		
		return arr;
	}

	private static void swap(int i, int j) {
		int temp = i;
		i = j;
		j = temp;
	}

}

	public static Path copyFileToDestFolder(Path source, Path destination) {

		if (destination.toFile().isFile()) {
			System.out.println("Can't copy to File!");
			return null;
		}

		Path copy = null;
		String d = destination.toFile() + File.separator + source.getFileName().toString();
		destination = Paths.get(d);

				destination = appendToPathName(destination.toFile(), " copy", ".txt");
				System.out.println("Destination renamed to :" + destination);

			} catch (IOException e) {

				System.out.println(e);

			}

		}

		return copy;

	}

	private static Path copyByRename(Path source, Path destinamtion) {
		System.out.println("Source Found: " + source.toFile().exists());

		Path copy = null;
		for (int i = 0; i < 10; i++) {
			try {

				copy = Files.copy(source, destinamtion, StandardCopyOption.COPY_ATTRIBUTES);
				if (copy != null) {
					System.out.println("Copy successfull!");
					break;
				}

			} catch (FileAlreadyExistsException e) {

				destinamtion = appendToPathName(destinamtion.toFile(), " copy", ".txt");
				System.out.println("Destination renamed to :" + destinamtion);

			} catch (IOException e) {

				System.out.println(e);

			}

		}

		return copy;
	}

	private static Path appendToPathName(File file, String concat, String extention) {
		String name = file.getName();

		// Extract just RAW name
		String justName = name.substring(0, name.lastIndexOf('.'));

		// Change the extracted name
		String newName = justName.concat(concat);

		// Add newName to parent path
		String newFilePath = file.getParent() + File.separator + newName + extention;

		File newFile = new File(newFilePath);

		return Paths.get(newFilePath);
	}

}

		int arr[] = { 5, 45, 64, 34, 2, 3 };
		
		for (int i = 0; i < arr.length; i++) { // 0 - L-1
			int at = findLargeAt(arr, i,arr.length); // 0,L-1
	 		swap(arr, i, at); 
		}
		System.out.println(Arrays.toString(arr));

	
	}

	private static void swap(int[] arr, int i, int j) {
		int temp = arr[i];
		arr[i] = arr[j];
		arr[j] = temp;
		
	}

	private static int findLargeAt(int[] arr, int from, int to) {
		int lg = Integer.MIN_VALUE; 
		int lgIndex = 0;
		for (int j = from; j < to; j++) { // 0-
			//System.out.print(arr[j]+", ");
			if(arr[j] > lg ){
				lgIndex = j;
				 lg = arr[j];
			}
		}

		return lgIndex;
	}
